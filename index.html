<body class="bg-gray-100 text-gray-800">

  <div class="max-w-3xl mx-auto py-10 px-4">

    <h1 class="text-3xl font-bold mb-6 text-center">🏅 Hello, Godot! 成就牆</h1>

    <p class="mb-6 text-center">勾選你已經達成的成就吧！系統會自動幫你保存進度。</p>


    <div id="achievement-list" class="grid gap-4">

      <!-- 成就項目將由 JavaScript 插入 -->

    </div>

  </div>


  <script>

    const achievements = [

      { key: "hello", title: "Hello, Godot!", desc: "第一次使用 Godot" },

      { key: "upload", title: "Uploaded!", desc: "成功提交遊戲" },

      { key: "demo", title: "Demo", desc: "參與期中展示" },

      { key: "partner", title: "Partner", desc: "與隊友合作" },

      { key: "solo", title: "Solo", desc: "全程獨立完成" },

      { key: "social", title: "Social", desc: "連續留言七天以上" },

      { key: "first", title: "First Finish!", desc: "第一次完成作品" },

      { key: "complete", title: "Complete", desc: "做出一款可玩遊戲" },

      { key: "challenge", title: "Challenge", desc: "完成創意挑戰" },

      { key: "attendance", title: "Full Attendance", desc: "每天打卡 / 留言" },

      { key: "survivor", title: "Survivor", desc: "完整參與整個 Jam" },

    ];


    const listContainer = document.getElementById("achievement-list");


    achievements.forEach(({ key, title, desc }) => {

      const isChecked = localStorage.getItem(`achieve_${key}`) === "true";

      const item = document.createElement("label");

      item.className = "flex items-start bg-white shadow px-4 py-3 rounded cursor-pointer hover:bg-gray-50";


      item.innerHTML = `

        <input type="checkbox" class="mt-1 mr-3" id="${key}" ${isChecked ? "checked" : ""} />

        <div>

          <div class="font-semibold text-lg">${title}</div>

          <div class="text-sm text-gray-600">${desc}</div>

        </div>

      `;


      const checkbox = item.querySelector("input");

      checkbox.addEventListener("change", () => {

        localStorage.setItem(`achieve_${key}`, checkbox.checked);

      });


      listContainer.appendChild(item);

    });

  </script>
